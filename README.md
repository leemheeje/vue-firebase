## 파이어베이스 컨피그 생성

```
파이어베이스 컨피그를 붙혀넣기해주세요
export default {
	apiKey: "",
	authDomain: "",
	databaseURL: "",
	projectId: "",
	storageBucket: "",
	messagingSenderId: "",
	appId: "",
	measurementId: "",
};
```
